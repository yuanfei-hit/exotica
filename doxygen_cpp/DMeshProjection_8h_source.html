74  virtual void project(const ompl::base::State *state,
75  ompl::base::EuclideanProjection &projection) const
76  {
77  // HIGHLIGHT_NAMED("DMeshProjection", "Calling project");
78  Eigen::VectorXd q(space_->getDimension());
79  compound_ ? space_->as<OMPLSE3RNStateSpace>()->OMPLStateToEigen(state, q) : space_->as<OMPLStateSpace>()->copyFromOMPLState(state, q);
80  scene_->Update(q);
81  ArrayFrame poses = scene_->getSolver().Solution->Phi;
82  for (std::size_t i = 0; i < dist_index_.size(); ++i)
83  projection(i) = (poses(dist_index_[i].second).p - poses(dist_index_[i].first).p).Norm();
84  }
